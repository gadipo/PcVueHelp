REPLACE Example

Applies to

This program illustrates the use of the REPLACE instruction.

The function REPLACE_OLDNEW replaces all occurrences of the old sub-strings with the new.

The function REPLACE_OLDNEW_NUMBER replaces a specific number of occurrences of the old sub-string with the new.

SUB REPLACE_OLDNEW ()
 
'-------- Variables used by REPLACE;
 
  CONST CASESENSITIVE = 1;
 
  CONST NOTCASESENSITIVE = 0;
 
  DIM sInputString AS STR, sOldSubstring AS STR, sNewSubstring AS STR;
 
  DIM sResult AS STR;
 
  
 
'-------- Initialize variables  
 
  sInputString = "The quick brown fox jumped over THE lazy dog";
 
  sOldSubstring = "THE";
 
  sNewSubstring = "a";
 
'-------- Non case sensitive replace  
 
  sResult = REPLACE(sInputString, sOldSubstring, sNewSubstring, NOTCASESENSITIVE);
 
  PRINT("Result from non-case sensitive replace: ", sResult);
 
'-------- Case sensitive replace 
 
  sResult = REPLACE(sInputString, sOldSubstring, sNewSubstring, CASESENSITIVE);
 
  PRINT("Result from case sensitive replace: ", sResult);
 
End Sub
 
 
 
SUB RELACE_OLDNEW_NUMBER ()
 
'-------- Variables used by REPLACE
 
  CONST CASESENSITIVE = 1;
 
  CONST NOTCASESENSITIVE = 0;
 
  DIM sInputString AS STR, sOldSubstring AS STR, sNewSubstring AS STR;
 
  DIM sResult AS STR;
 
  DIM iOccurrenceNumber AS INTEGER;
 
'-------- Initialize variables  
 
  sInputString = "The quick brown fox jumped over the lazy dog";
 
  sOldSubstring = "the";
 
  sNewSubstring = "a";
 
'-------- Replace only one occurrence
 
  iOccurrenceNumber = 1;
 
  sResult = REPLACE(sInputString, sOldSubstring, sNewSubstring, NOTCASESENSITIVE, iOccurrenceNumber);
 
  PRINT("Replace one occurrence of 'the': ", sResult);
 
'-------- Replace 2 occurences 
 
  iOccurrenceNumber = 2;
 
  sResult = REPLACE(sInputString, sOldSubstring, sNewSubstring, NOTCASESENSITIVE, iOccurrenceNumber);
 
  PRINT("Replace two occurrences of 'the': ", sResult);
 
End Sub
