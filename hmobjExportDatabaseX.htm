Export a Database

Declare internal variables:

Dim ExportFormat As HDSMgtLib.HDSMgtExportFormat

Dim ExportType As Integer

Pause database recording:

Private Sub Text15_Click()

Dim dbMgt As New HDSMgt

Dim DB As HDSMgtDatabase

    Set DB = dbMgt.Databases("DefaultDB")

    DB.PauseRecording

End Sub

Set the export format to Text:

Private Sub Text13_Click()

    ExportFormat = HIS_SRV_EXPORT_TXT

    DBexport

End Sub

Export the database records to C:\temp:

Private Sub DBexport()
Dim dbMgt As New HDSMgt
Dim DB As HDSMgtDatabase

    Set DB = dbMgt.Databases("DefaultDB")

    Call DB.StartExport(PurgeDateStart, _

      PurgeDateEnd, "c:\temp", "dat", _

      HIS_SRV_EXPORT_TXT, True, ";", "Null")

End Sub
