ExProperties Iteration Example

'-------- Retrieve and iterate an ExProperties collection --------
Private Sub ExPropertiesEx1_Click()
Dim ExProps As VarExProperties
Dim ExProp As VarExProperty

' Retrieve the Extended Properties collection of the variable Bit1

Set ExProps = TheseVariables("Bit1").ExProperties

' Check that the collection is not empty
If ExProps.Count > 0 Then
' Iterate the collection
For I = 1 To ExProps.Count
' Retrieve the Extended Property according to the rank
' Note that ExProps(I) is equivalent to ExpProps.Item(I)
Set ExProp = ExProps(I)
' Check that the Extended Property can be read
If (ExProp.Access And fvPropAccessRead) = fvPropAccessRead Then
' Display the Name and Value
MsgBox ExProp.Name & " = " & ExProp.Value
End If
Next
End If

End Sub
