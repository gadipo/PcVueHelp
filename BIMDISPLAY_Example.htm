BIMDISPLAY Example

Applies to

This program illustrates the use of the BIMDISPLAY instruction.

Illustrations in this book are based on an Ifc model provided by the XBIM team and available under the Common Development and Distribution License (CDDL).
The original "Sample house" model can be downloaded at https://github.com/xBimTeam/XbimEssentials/.
Credits: The XBIM team.

Example prerequisites

  • A mimic containing a BIM control and a List-box control must be open. Show picture

  • The BIM control is used to visualize the contents of a .Ifc file. This example requires the file to be located in the project's BIM folder.

  • A text variable, @SB_EXAMPLES.PresetName, is required. This variable must have the Command property set.

  • A register variable, @SB_EXAMPLES.LabelValue, is required. This variable must have the Command property set.

  • An Xml file is used to save and load the presets. This file is created the first time the SAVEPRESETS mode is used. In this example it is located in project's BIM folder.

DIM sWindow AS STR;
 
DIM sBranch AS STR;
 
DIM sIdentity AS STR;
 
DIM sPresetListBox AS STR;
 
DIM sPresetFileName AS STR;
 
 
 
'-------- Must be ran once to initialize variables
 
SUB Main ()
 
  sWindow = "Mimic3"; ' Name of the mimic comtaining the BIM control
 
  sBranch = "";
 
  sIdentity = "BimControl1"; 'Identity of the BIM control
 
  sPresetListBox = "AIListBox1"; 'Identity of the list box used to display presets
 
  sPresetFilename = "UserPresets.xml"; 'Name of the file used to save presets
 
END SUB
 
 
 
'-------- Load an Ifc file in the BIM control. File name passed as an argument.
 
SUB Exec_LoadIFC ()
 
  DIM lReturn AS LONG;
 
  DIM cIfcFileName AS STR;
 
  cIfcFileName = GETARG("ARG1");
 
  lReturn = BIMDISPLAY("LOADIFC", sWindow, sBranch, sIdentity, cIfcFileName);
 
  PRINT("Result from mode LOADIFC = ", lReturn);
 
END SUB
 
 
 
'-------- Open the BIM control Ifc properties dialog
 
Sub Exec_OpenPropertiesWindow ()
 
  DIM lReturn AS LONG;
 
  lReturn = BIMDISPLAY("OPENPROPERTIESWINDOW", sWindow, sBranch, sIdentity);
 
  PRINT("Result from mode OPENPROPERTIESWINDOW = ", lReturn);
 
END SUB
 
 
 
'-------- Add a user preset. The preset name comes from the variable @SB_EXAMPLES.PresetName
 
SUB Exec_AddPreset()
 
  DIM lReturn AS LONG;
 
  DIM cPresetName AS STR;
 
  CONST CAMERA = 2; 'To add a camera preset
 
  CONST CUTPLANE = 1; 'To add a cutplane preset
 
  CONST BOTH = 0; 'To add both a camera and cutplane preset
 
  cPresetName = @SB_EXAMPLES.PresetName;
 
  lReturn = BIMDISPLAY("ADDPRESET", sWindow, sBranch, sIdentity, cPresetName, CAMERA);
 
  PRINT("Result from mode ADDPRESET = ", lReturn);
 
END SUB
 
 
 
'-------- Go to a preset. The preset name comes from the list box populated by Exec_GetPresets
 
SUB Exec_GoToPreset()
 
  DIM lReturn AS LONG;
 
  DIM cPresetName AS STR;
 
  DIM iIndex AS Integer;
 
  iIndex = TOI(LISTBOX("GETSELECTEDINDEX", sWindow, sBranch, sPresetListBox));
 
  cPresetName = LISTBOX("GETUSERDATA", sWindow, sBranch, sPresetListBox, iIndex);
 
  @SB_EXAMPLES.PresetName = cPresetName;
 
  lReturn = BIMDISPLAY("GOTOPRESET", sWindow, sBranch, sIdentity, cPresetName);
 
  PRINT("Result from mode GOTOPRESET ", cPresetName, " = ", lReturn);
 
END SUB
 
 
 
'-------- Go to a preset. The preset name is passed as an argument
 
SUB Exec_GoToStandardPreset()
 
  DIM lReturn AS LONG;
 
  DIM cPresetName AS STR;
 
  cPresetName = GETARG("ARG1");
 
  lReturn = BIMDISPLAY("GOTOPRESET", sWindow, sBranch, sIdentity, cPresetName);
 
  PRINT("Result from mode GOTOPRESET ", cPresetName, " = ", lReturn);
 
END SUB
 
 
 
'-------- Get the BIM control user added presets and display them in a list box
 
SUB Exec_GetPresets()
 
  DIM iPresetCount AS Integer, iIndex AS Integer, i AS Integer;
 
  CONST INDEXEND = -1; ' Used AS the ListBox index adds to the end of the list.
 
  DIM cText AS STR, cUserData AS STR;
 
  ListBox("CLEAR", sWindow, sBranch, sPresetListBox);
 
  iPresetCount = TOI(BIMDISPLAY("GETPRESETCOUNT", sWindow, sBranch, sIdentity));
 
  PRINT(iPresetCount, " presets found");
 
  FOR (iIndex=0; iIndex < iPresetCount; iIndex++)
 
    cText = BIMDISPLAY("GETPRESETNAME", sWindow, sBranch, sIdentity, iIndex);
 
    cUserdata = cText;
 
    ListBox("INSERT", sWindow, sBranch, sPresetListBox, cText, cUserData, INDEXEND); 
 
  NEXT
 
END SUB
 
 
 
'-------- Save BIM Control user added presets in an Xml file
 
SUB Exec_SavePresets ()
 
  DIM lReturn AS LONG;
 
  lReturn = BIMDISPLAY("SAVEPRESETS", sWindow, sBranch, sIdentity, sPresetFileName);
 
  PRINT("Result from mode SAVEPRESETS = ", lReturn);
 
END SUB
 
 
 
'-------- Load BIM control user added presets from an Xml file
 
SUB Exec_LoadPresets ()
 
  DIM lReturn AS LONG;
 
  CONST ADDPRESETS = 1; 'Adds presets. Using this sub-mode presets may be duplicated.
 
  CONST REPLACEPRESETS = 0; 'Adds presets replacing any already existing that have the same name.
 
  lReturn = BIMDISPLAY("LOADPRESETS", sWindow, sBranch, sIdentity, sPresetFileName, REPLACEPRESETS);
 
  PRINT("Result from mode LOADPRESETS = ", lReturn);
 
END SUB
 
 
 
'-------- Show/hide object according to its type. Show(1) or hide(0) is passed as an argument.
 
SUB  Exec_ShowObjectByType()
 
  DIM lReturn AS LONG;
 
  DIM iShow AS INTEGER;
 
  DIM sIfcTypeValue AS STR;
 
  iShow = IVAL(GETARG("ARG1"));
 
  sIfcTypeValue = @SB_EXAMPLES.ObjectType;
 
  lReturn = BIMDISPLAY("SHOWOBJECTBYTYPE", sWindow, sBranch, sIdentity, sIfcTypeValue, iShow);
 
  PRINT("Result from mode SHOWOBJECTBYTYPE = ", lReturn);
 
END SUB
 
 
 
'-------- Zoom in on a specifc object using its unique LabelValue. The object is highlighted and the other Ifc elements are set to transparent.
 
SUB Exec_ZoomObject()
 
  DIM lReturn AS LONG;
 
  DIM iIfcLabelValue AS INTEGER;
 
  CONST SELECTOBJECT = 1;
 
  CONST CHANGEMODEOPACITY = 1;
 
  iIfcLabelValue = TOI(@SB_EXAMPLES.LabelValue);
 
  lReturn = BIMDISPLAY("ZOOMOBJECT", sWindow, sBranch, sIdentity, iIfcLabelValue, SELECTOBJECT, CHANGEMODEOPACITY );
 
  PRINT("Result from mode ZOOMOBJECT = ", lReturn);
 
END SUB
