ExProperties Example Display Access Properties

'-------- Display the Access property of an Extended Property --------
'by using fvPropAccessNone, fvPropAccessRead and fvPropAccessWrite
'as binary filters to check the value of the Access property

Private Sub Text1_Click()

Dim va As VarExProperty
Set va = Variables("Bit1").ExProperties("TATT3")

If va.Access = fvPropAccessNone Then
'value 0
MsgBox "No Access"
End If

'THIS NEXT TEST IS INCORRECT BECAUSE IT ONLY WORKS FOR READ-ONLY ATTRIBUTE
If va.Access = fvPropAccessRead Then
'value 1
MsgBox "Access Read-only"
End If

'THESE TESTS ARE CORRECT BECAUSE THEY ALSO CHECK THE ACCESS PROPERTY
If va.Access And fvPropAccessRead Then
MsgBox "Access Read"
End If

If va.Access And fvPropAccessWrite Then
'value 2
MsgBox "Access Write"
End If

If va.Access = (fvPropAccessRead Or fvPropAccessWrite) Then
'value 3 (i.e. 1+2)
MsgBox "Access Read - Write"
End If

End Sub
