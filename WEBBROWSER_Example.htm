WEBBROWSER Example

Applies to

This example contains a set of functions to script the Web browser control using the WEBBROWSER instruction.

Sub NavigateTo()

  Dim lRet as Long;

  Dim sUri as Str;

 

  'ARG1 is a valid Uri such as https://example.com or file://C:\MyFolder\MyFile.pdf

  'ARG1 can also be an absolute filename such as C:/MyFolder/MyFile.pdf

  sUri = GETARG("ARG1");

 

  lRet = WebBrowser("NAVIGATE", "WebBrowserMimic", "", "WebBrowserCtrl", sUri);

  Print("Navigate to ", sUri, " executed with result: ",TOC(lRet));

End Sub

 

Sub NavigateBack()

  Dim lRet as Long;

 

  lRet = WebBrowser("BACK", "WebBrowserMimic", "", "WebBrowserCtrl");

  Print("Navigate back executed with result: ",TOC(lRet));

End Sub

 

Sub NavigateForward()

  Dim lRet as Long;

 

  lRet = WebBrowser("FORWARD", "WebBrowserMimic", "", "WebBrowserCtrl");

  Print("Navigate forward executed with result: ",TOC(lRet));

End Sub

 

Sub GetZoomFactor()

  Dim lRet as Long;

 

  lRet = WebBrowser("GETZOOMFACTOR", "WebBrowserMimic", "", "WebBrowserCtrl");

  Print("Zoom factor is: ",TOC(lRet));

End Sub

 

Sub GetInfo()

  Dim sRet as Str;

 

  sRet = WebBrowser("GETSOURCEINFO", "WebBrowserMimic", "", "WebBrowserCtrl",0);

  Print("The page Url is: ",sRet);

  sRet = WebBrowser("GETSOURCEINFO", "WebBrowserMimic", "", "WebBrowserCtrl",1);

  Print("The document title is: ",sRet);

  sRet = WebBrowser("GETSOURCEINFO", "WebBrowserMimic", "", "WrongCtrlId",1);

  Print("The document title if the Web ctrl Id is wrong is: ",sRet);

End Sub

 

Sub PrintToPdf()

  Dim lRet as Long;

 

  lRet = WebBrowser("PRINTTOPDF", "WebBrowserMimic", "", "WebBrowserCtrl", "PrintToPdf.pdf",1);

  Print("PrintToPdf executed with result: ",TOC(lRet));

End Sub
