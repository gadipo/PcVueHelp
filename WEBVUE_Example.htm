Example 1 - Data returned by WEBVUE mode LIST - Syntax 1

The following example uses the mode LIST (syntax 1 text-based) to list all users connected via a Web client.

Sub ListSyntax1()

Dim iRet as Integer;

Dim strClientList as Str;

Dim hBuff as Long;

 

hBuff = ALLOC_BUFFER(5000); 'Allocate a buffer of 5 kb

iRet = Webvue("LIST", hBuff); 'Retrieve the list of sessions

Print("Mode LIST Syntax 1 executed with result: ", iRet);

 

if (iRet == -1) then 'List retrieval failed

Print("List retrieval failed - Memory buffer is too small");

else 'List retrieval is successful

strClientList = CGET_BUFFER(hBuff,0,255);

Print("Number of web clients: ", iRet);

Print("List of web clients: ", strClientList);

end if

 

FREE_BUFFER(hBuff);

End Sub

Example 2 - Data returned by WEBVUE mode LIST - Syntax 6

The following example using the mode LIST (syntax 6 XML-based) to list all users connected via a Web client.

Typical XML fragment returned by mode LIST - Syntax 6

<list myid="767565456788">

<session>

id="535625422312"

creationtime=xxxxxxx

username="User01"

clienttype="webvue"

clientip:xxx.xxx.x.xxx

clienthost: MM-MBP.local

useragent: Mozilla/5(Maintosh; intel Mac OS X 10_15_7)AppleWebKit/605.1.15 (KHTML, like Gecko) Version/15.3 Safari/

browser: Safari

os: Mac

devicetype: Desktop

</session>

</list>

Script to analyze the returned data

SUB LISTWEBVUESESSIONS()

DIM iRet as Integer;

DIM hBuff as Long;

DIM i As Integer;

DIM iSessionCount As Integer;

DIM sPath As Str;

 

Webvue("LIST"); 'Retrieve the list of sessions

Print("Mode LIST Syntax 6 executed");

 

iSessionCount = XMLPATH("COUNT","webvue/list","list/session"); 'Retrieve the list of sessions in the XML fragment

Print(iSessionCount, " Web session(s) currently active.");

 

hBuff = XMLPATH("GET","webvue/list","list.myid");

If (hBuff != 0 ) Then

Print("myid - The session Id of the script caller is: ",CGET_BUFFER ( hBuff, 0, 255 ));

End If

 

If (iSessionCount>0) then

For (i=1; i<=iSessionCount;i++) 'Iterate the collection of session elements in the XML fragment

sPath = FORMAT("list.session[%d].id", i);

hBuff = XMLPATH( "GET", "webvue/list", sPath );

If (hBuff != 0) Then

Print("Web session number ",i);

Print("\tid: ",CGET_BUFFER(hBuff,0,255));

 

sPath = FORMAT("list.session[%d].creationtime", i);

Print("\tcreationtime: ",DATETIMESTRING(XMLPATH("GETDOUBLE","webvue/list",sPath)));

 

sPath = FORMAT("list.session[%d].username",i);

hBuff = XMLPATH("GET","webvue/list",sPath );

Print("\tusername: ",CGET_BUFFER(hBuff,0,255));

 

sPath = FORMAT("list.session[%d].clienttype",i);

hBuff = XMLPATH ("GET","webvue/list",sPath);

Print("\tclienttype: ",CGET_BUFFER(hBuff,0,255));

 

sPath = FORMAT("list.session[%d].clientip",i);

hBuff = XMLPATH ("GET","webvue/list",sPath);

Print("\tclientip: ",CGET_BUFFER(hBuff,0,255));

 

sPath = FORMAT("list.session[%d].clienthost",i);

hBuff = XMLPATH ("GET","webvue/list",sPath);

Print("\tclienthost: ",CGET_BUFFER(hBuff,0,255));

 

sPath = FORMAT("list.session[%d].useragent",i);

hBuff = XMLPATH ("GET","webvue/list",sPath);

Print("\tuseragent: ",CGET_BUFFER(hBuff,0,255));

 

sPath = FORMAT("list.session[%d].browser",i);

hBuff = XMLPATH ("GET","webvue/list",sPath);

Print("\tbrowser: ",CGET_BUFFER(hBuff,0,255));

 

sPath = FORMAT("list.session[%d].os",i);

hBuff = XMLPATH ("GET","webvue/list",sPath);

Print("\tos: ",CGET_BUFFER(hBuff,0,255));

 

sPath = FORMAT("list.session[%d].devicetype",i);

hBuff = XMLPATH ("GET","webvue/list",sPath);

Print("\tdevicetype: ",CGET_BUFFER(hBuff,0,255));

End If

Next

End If

End Sub
