SVBRANCH Example

Applies to

This program illustrates the use of the SVBRANCH instruction.

Example prerequisites

  • One or more variables in a branch called @SB_EXAMPLES are required. If changing the variable values manually the Command attribute must be set.

  • To set the variables and visualize the results of using the program, a mimic displaying the variable values and status is useful (using the Grid control in variable tracking mode for example).

'-------- Program wide variables
 
DIM sBranch As STR;
 
 
 
SUB Main ()
 
'-------- Initialize variables
 
  sBranch = "SB_EXAMPLES"; 'The name of the branch to use
 
END SUB
 
 
 
'-------- Disable all variables in the branch sBranch
 
SUB Exec_DISABLE()
 
  DIM iReturn AS INTEGER;
 
  iReturn = SVBRANCH("DISABLE", sBranch);
 
  IF(iReturn == 1) THEN
 
    PRINT("Disable branch ", sBranch, " successful");
 
  ELSE
 
    PRINT("Disable branch ", sBranch, " failed");
 
  END IF
 
END SUB
 
 
 
'-------- Enable all variables in the branch sBranch
 
SUB Exec_ENABLE()
 
  DIM iReturn AS INTEGER;
 
  iReturn = SVBRANCH("ENABLE", sBranch);
 
  IF(iReturn == 1) THEN
 
    PRINT("Enable branch ", sBranch, " successful");
 
  ELSE
 
    PRINT("Enable branch ", sBranch, " failed");
 
  END IF
 
END SUB
 
 
 
'-------- Mask all variables in the branch sBranch, using a mask level passed as an argument
 
SUB Exec_MASK ()
 
  DIM iReturn AS INTEGER,iLevel As INTEGER;
 
'-------- Get mask passsed as an argument  
 
  iLevel = IVAL(GETARG("ARG1"));
 
  iReturn = SVBRANCH("MASK", sBranch, iLevel);
 
  IF(iReturn == 1) THEN
 
    PRINT("Mask level ", iLevel, " successful");
 
  ELSE
 
    PRINT("Mask level ", iLevel, " failed");
 
  END IF
 
END SUB
 
 
 
'-------- Unmask all variables in the branch sBranch, using a mask level passed as an argument
 
SUB Exec_UNMASK ()
 
  DIM iReturn AS INTEGER,iLevel As INTEGER;
 
'-------- Get mask passsed as an argument  
 
  iLevel = IVAL(GETARG("ARG1"));
 
  iReturn = SVBRANCH("UNMASK", sBranch, iLevel);
 
  IF(iReturn == 1) THEN
 
    PRINT("UnMask level ", iLevel, " successful");
 
  ELSE
 
    PRINT("UnMask level ", iLevel, " failed");
 
  END IF
 
END SUB
 
 
 
'-------- Temporarily change the alarm level of all alarms in the branch sBranch, using an alarm level passed as an argument
 
SUB Exec_SETALARMLEVEL ()
 
  DIM iReturn AS INTEGER,iLevel As INTEGER;
 
'-------- Get new alarm level passsed as an argument  
 
  iLevel = IVAL(GETARG("ARG1"));
 
  iReturn = SVBRANCH("SETALARMLEVEL", sBranch, iLevel);
 
  IF(iReturn == 1) THEN
 
    PRINT("Set alarm level ", iLevel, " successful");
 
  ELSE
 
    PRINT("Set alarm level ", iLevel, " failed");
 
  END IF
 
END SUB
 
 
 
'-------- Restore temporary alarm levels of all alarms in the branch sBranch (level set with SETALARMLEVEL)
 
SUB Exec_RESTOREALARMLEVEL ()
 
  DIM iReturn AS INTEGER;
 
  iReturn = SVBRANCH("RESTOREALARMLEVEL", sBranch);
 
  IF(iReturn == 1) THEN
 
    PRINT("Restore alarm level successful");
 
  ELSE
 
    PRINT("Restore alarm level failed");
 
  END IF
 
END SUB
 
 
 
'-------- Toggle all alarms in the branch sBranch, from ON to OFF to ON 
 
SUB Exec_RESETANDSETALARMBRANCH ()
 
  DIM iReturn AS INTEGER;
 
  iReturn = SVBRANCH("RESETANDSETALARM", sBranch);
 
  IF(iReturn == 1) THEN
 
    PRINT("Reset and set alarm branch ",sBranch, " successful");
 
  ELSE
 
    PRINT("Reset and set alarm branch ",sBranch, " failed");
 
  END IF
 
END SUB
 
 
 
'-------- Toggle all alarms that pass the sFilter filter, from ON to OFF to ON 
 
SUB Exec_RESETANDSETALARMFILTER ()
 
  DIM iReturn AS INTEGER;
 
  DIM sFilter As STR;
 
  sFilter = "=(#A1 == DOM1)";
 
  iReturn = SVBRANCH("RESETANDSETALARM", "", sFilter);
 
  IF(iReturn == 1) THEN
 
    PRINT("Reset and set alarm with filter ",sFilter, " successful");
 
  ELSE
 
    PRINT("Reset and set alarm with filter ",sFilter, " failed");
 
  END IF
 
END SUB
