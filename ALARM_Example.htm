ALARM Example

Applies to

This program illustrates the use of the ALARM instruction.

Example prerequisites

  • A text variable, @SB_EXAMPLES.AlarmName, is required. This variable must have the Command property set.

  • One or more bit variables with the Alarm property set. These variables must also have the Command property set.
  • A mimic containing a Send-Text animation to set the variable SB_EXAMPLES. To use the ACKTAG example this mimic will also need an Alarm Viewer. Show picture

DIM sWindow AS Str, sBranch AS Str, sIdentity AS STR;
 
'-------- Intialise program wide variables
 
SUB Main()
 
  sWindow = "Mimic8";
 
  sBranch = "";
 
  sIdentity = "Alarm1";
 
END SUB
 
 
 
'-------- Exercises the modes of Syntax 1. The mode is passed as an argument.
 
SUB EXEC_ALARM_SYNTAX1 ()
 
  DIM sAlarmName AS STR;
 
  DIM sMode AS STR;
 
  DIM iReturn AS INTEGER;
 
  sMode = GETARG("ARG1"); ' Get mode passed as an argument
 
  sAlarmName = @SB_EXAMPLES.AlarmName;
 
  iReturn = ALARM(sMode, sAlarmName);
 
  Print("Syntax 1, Mode ",sMode, " returned ",iReturn);
 
END SUB
 
 
 
'-------- Exercises the modes of Syntax 5. The mode is passed as an argument.
 
SUB EXEC_ALARM_SYNTAX5 ()
 
  DIM sAlarmName AS STR;
 
  DIM sMode AS STR;
 
  DIM iReturn AS INTEGER;
 
  CONST Flag = 1;
 
  sMode = GETARG("ARG1"); ' Get mode passed as an argument
 
  sAlarmName = @SB_EXAMPLES.AlarmName;
 
  iReturn = ALARM(sMode, sAlarmName, Flag);
 
  Print("Syntax 5, Mode ",sMode, " returned ",iReturn);
 
END SUB
 
 
 
'-------- Acknowledge all alarms. Use carefully - may affect system performance if there are many alarms.
 
SUB EXEC_ALARM_ACKALL ()
 
  DIM iReturn AS INTEGER;
 
  CONST Flag = 1; 'Indicates that the action should be logged as it would be from an Alarm Viewer
 
  sAlarmName = @SB_EXAMPLES.AlarmName;
 
  iReturn = ALARM("ACKALL", Flag); 
 
  Print("Acknowledge all alarms returned ",iReturn);
 
END SUB
 
 
 
'-------- Acknowledge a single alarm using it's ID within an Alarm Viewer
 
SUB EXEC_ALARM_ACKTAG ()
 
  DIM iReturn As INTEGER;
 
  DIM sAlarmId AS STR;
 
  sAlarmId = ALARMDISPLAY("SELECTED", sWindow, sBranch, sIdentity); ' Get selected alarm ID
 
  iReturn = ALARM("ACKTAG", sAlarmId);
 
  IF (iReturn == 1) THEN
 
    PRINT("Acknowledge alarm ID ", sAlarmID, " successful");
 
  ELSE
 
    PRINT("Acknowledge alarm ID ", sAlarmID, " failed");
 
  END IF
 
END SUB
 
 
 
'-------- Acknowledge alarms by domain and nature
 
SUB EXEC_ALARM_ACKDOMNAT ()
 
  DIM sDomain AS STR, sNature AS STR;
 
  DIM iReturn AS INTEGER;
 
  CONST Flag = 1; 'Indicates that the action should be logged as it would be from an Alarm Viewer
 
  sDomain = "DOM1";
 
  sNature = "";
 
  iReturn = ALARM("ACKDOMNAT", sDomain, sNature, Flag);
 
  Print("Acknowledge alarms of Domain ", sDomain, " and Nature ", sNature, " returned ",iReturn);
 
END SUB
 
 
 
'-------- Acknowledge alarms by priority
 
SUB EXEC_ALARM_ACKPRIO ()
 
  DIM iPriority AS INTEGER;
 
  DIM iReturn AS INTEGER;
 
  CONST Flag = 1; 'Indicates that the action should be logged as it would be from an Alarm Viewer
 
  iPriority = 1;
 
  iReturn = ALARM("ACKPRIO", iPriority, Flag);
 
  Print("Acknowledge alarms of Priority ", iPriority, " returned ",iReturn);
 
END SUB
 
 
 
'-------- Temporarily change an alarms level
 
SUB EXEC_ALARM_SETALARMLEVEL ()
 
  DIM iReturn As INTEGER, iLevel AS INTEGER;
 
  DIM sAlarmName AS STR;
 
  sAlarmName = @SB_EXAMPLES.AlarmName;
 
  iLevel = IVAL(GETARG("ARG1")); ' Get new alarm level passed as argument
 
  iReturn = ALARM("SETALARMLEVEL", sAlarmName, iLevel);
 
  Print("Change alarm level of ", sAlarmName, " to level ", iLevel, " returned ",iReturn);
 
END SUB
