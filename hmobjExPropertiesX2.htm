ExProperties Example to Write an Extended Property

'-------- Retrieve and write a particular Extended Property --------
Private Sub ExPropertiesEx2_Click()
Dim ExProps As VarExProperties
Dim ExProp As VarExProperty

' Retrieve the Extended Properties collection of the variable "Bit1"
Set ExProps = TheseVariables("Bit1").ExProperties

' Check that the collection is not empty
If ExProps.Count > 0 Then
' Retrieve the Extended Property "TATT4"
Set ExProp = ExProps("TATT4")
If (ExProp.Access And fvPropAccessWrite) = fvPropAccessWrite Then
' Write the Extended Property value
ExProp.Value = "The name is TATT4"
End If
End If

End Sub
