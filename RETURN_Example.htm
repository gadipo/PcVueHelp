In this example, the Main module passes values to the Factorial module and prints the returned values.

SUB Main ()
DIM J As Integer;
For (J=1; J<=10; J++)
Print("Factorial ", J, " = ", Factorial(J));
Next J;
END SUB

 

SUB Factorial (Input)
DIM I As Integer;
DIM Result As Long;
Result = 1;
If (Input < 1) Then
Return (0);
End If
For ( I = 1; I <=Input; I++)
Result = Result*TOL(I);
Next I;
Return(Result);
END SUB
