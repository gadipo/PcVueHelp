OPC Example

Applies to

This program illustrates the use of the OPC instruction.

Example prerequisites

  • An OPC Server configured with the name Server01

  • An OPC Group configured with the name Group01 under the server Server01.

DIM sServerAlias AS STR, sGroupName AS STR, sServerAlias.GroupName As Str;
 
 
 
'-------- Intitialize program wide variables
 
SUB Main ()
 
  sServerAlias = "Server01";
 
  sGroupName = "Group01";
 
  sServerAlias.GroupName = ADDSTRING(sServerAlias, ".", sGroupName);
 
  PRINT("ServerAlias.GroupName is ",sServerAlias.GroupName);
 
END SUB
 
 
 
'-------- Force a read of an OPC group
 
Sub Exec_OPC_READ()
 
  OPC("READ", sServerAlias.GroupName, "DEVICE" );
 
End Sub
 
 
 
'-------- Start an OPC Group
 
SUB Exec_OPC_STARTGROUP()
 
  OPC("STARTGROUP", sServerAlias.GroupName);
 
END SUB
 
 
 
'-------- Stop an OPC Group
 
SUB Exec_OPC_STOPGROUP()
 
  OPC("STOPGROUP", sServerAlias.GroupName);
 
END SUB
 
 
 
'-------- Check OPC group status
 
SUB CHECKSTATUS()
 
  DIM sVarName AS STR;
 
  DIM iValue As INTEGER;
 
'-------- Construct status var name and use deferred addressing to get value
 
  sVarName = ADDSTRING("System.OPC.", sServerAlias,".",sGroupName, ".On");
 
  iValue = ?sVarName;
 
  IF(iValue == 1) Then
 
    PRINT("OPC Group ", sServerAlias, ".",sGroupName, " is running");
 
  ELSE
 
    PRINT("OPC Group ", sServerAlias, ".",sGroupName, " is stopped");
 
  END IF
 
END SUB
 
 
 
'-------- Temporarily set specific trace bits. The state of other trace bits is unaffected. The effect is immediate.
 
Sub Exec_SETTRACE ()
 
  DIM sHexValue AS STR;
 
  sHexValue = "00300000"; 'Hex equivalent of trace 21 and trace 22 set.
 
  OPC("SETTRACE", sServerAlias.GroupName, sHexValue);'
 
End Sub
 
 
 
'-------- Temporarily reset specific trace bits. The state of other trace bits is unaffected. The effect is immediate.
 
Sub Exec_RESETTRACE ()
 
  DIM sHexValue AS STR;
 
  sHexValue = "00300000"; 'Hex equivalent of trace 21 and trace 22 set.
 
  OPC("RESETTRACE", sServerAlias.GroupName, sHexValue);'
 
End Sub
 
 
 
'-------- Temporarily change (overwrite) all the trace bits. The effect is immediate.
 
Sub Exec_PUTMASKTRACE ()
 
  DIM sHexValue AS STR;
 
  sHexValue = "00300000"; 'Hex equivalent of trace 21 and trace 22 set.
 
  OPC("PUTMASKTRACE", sServerAlias.GroupName, sHexValue);
 
End Sub
 
 
 
'-------- Permanently change (overwrite) all the trace bits. Taken into account after restart.
 
Sub Exec_SAVEMASKTRACE ()
 
  DIM sHexValue AS STR;
 
  sHexValue = "00300000"; 'Hex equivalent of trace 21 and trace 22 set.
 
  OPC("SAVEMASKTRACE", sServerAlias.GroupName, sHexValue);
 
End Sub
 
 
 
'-------- Permanently reset (overwrite) all the trace bits. Taken into account after restart.
 
Sub Exec_RESETSAVEMASKTRACE ()
 
  DIM sHexValue AS STR;
 
  sHexValue = "00000000";
 
  OPC("SAVEMASKTRACE", sServerAlias.GroupName, sHexValue);
 
End Sub
