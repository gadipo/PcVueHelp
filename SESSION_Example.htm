'-------- Get information about a session
 
SUB Exec_SESSION_Info ()
 
  DIM iType AS INTEGER;
 
  PRINT("Session ID = ", SESSION("GETCURRENTID"));
 
  iType = SESSION("GETTYPE"); 'Get the session type
 
  IF(iType == 1) THEN
 
    PRINT("Session type = Desktop client"); 
 
  ELSE
 
    IF(iType == 100) THEN
 
      PRINT("Session type = WebVue");
 
    ELSE
 
      IF(iType == 101) THEN 
 
        PRINT("Session type = TouchVue");
 
      ELSE
 
        IF(iType == 102) THEN
 
          PRINT("Session type= SnapVue");
 
        ELSE
 
          IF(iTyep == 103) THEN
 
            PRINT("Session type = WebScheduler");
 
          Else 
 
            IF(iTyep == 104) THEN
 
              PRINT("Session type = Web Service Toolkit client");
 
            ELSE
 
              PRINT("Session type is unknown or error";
 
            END IF
 
          END IF
 
        END IF
 
      END IF
 
    END IF
 
  END IF
 
'-------- If it's a session in a web context then get further information
 
  IF (iType > 99) THEN
 
    PRINT("WebBrowser = ", SESSION("GETWEBBROWSER"));
 
    PRINT("Web client IP = ", SESSION("GETWEBCLIENTIP"));
 
    PRINT("Web device type = ", SESSION("GETWEBDEVICETYPE"));
 
    PRINT("Operating system = ", SESSION("GETWEBOS"));
 
  END IF
 
END SUB
