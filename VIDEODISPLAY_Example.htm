VIDEODISPLAY Example

Applies to

This program illustrates the use of the VIDEODISPLAY instruction.

Example prerequisites

  • A mimic containing a Video control must be open.

'-------- Program wide variables
 
DIM sWindow AS Str, sBranch AS STR, sIdentity AS STR, sURL AS STR;
 
 
 
SUB Main ()
 
'-------- Intialize program wide variables
 
  sWindow = "mimic5"; 'Mimic name
 
  sBranch = ""; ' Branch
 
  sIdentity = "VideoCtrl1"; 'Video control Id
 
  sURL = "rtsp://[username:password@]ip_address[:rtsp_port]/server_URL[[?param1=val1[?param2=val2]…[?paramN=valN]]"; 'Video stream URL. Customize value to suite your specific setup (credentials, port...)
 
END SUB
 
 
 
'-------- Play the specific video stream
 
SUB Exec_PLAY ()
 
  DIM lReturn AS LONG;
 
  lReturn = VIDEODISPLAY("PLAY", sWindow, sBranch, sIdentity, sURL);
 
  IF (lReturn == 1) THEN
 
    PRINT("VIDEODISPLAY mode PLAY successful");
 
  ELSE
 
    PRINT("VIDEODISPLAY mode PLAY failed");
 
  END IF
 
END SUB
 
 
 
'-------- Stop the specific video stream
 
SUB Exec_STOP ()
 
  DIM lReturn AS LONG;
 
  lReturn = VIDEODISPLAY("STOP", sWindow, sBranch, sIdentity);
 
  IF (lReturn == 1) THEN
 
    PRINT("VIDEODISPLAY mode STOP successful");
 
  ELSE
 
    PRINT("VIDEODISPLAY mode STOP failed");
 
  END IF
 
END SUB
 
 
 
'-------- Pause video stream. Note that pause has a toggle action.
 
SUB Exec_PAUSE ()
 
  DIM lReturn AS LONG;
 
  lReturn = VIDEODISPLAY("PAUSE", sWindow, sBranch, sIdentity);
 
  IF (lReturn == 1) THEN
 
    PRINT("VIDEODISPLAY mode PAUSE successful");
 
  ELSE
 
    PRINT("VIDEODISPLAY mode PAUSE failed");
 
  END IF
 
END SUB
 
 
 
'-------- Take a snapshot of the video stream
 
SUB Exec_SNAPSHOT ()
 
  DIM lReturn AS LONG;
 
  lReturn = VIDEODISPLAY("SNAPSHOT", sWindow, sBranch, sIdentity);
 
  IF (lReturn == 1) THEN
 
    PRINT("VIDEODISPLAY mode SNAPSHOT successful - check project's TP folder!");
 
  ELSE
 
    PRINT("VIDEODISPLAY mode SNAPSHOT failed");
 
  END IF
 
END SUB
