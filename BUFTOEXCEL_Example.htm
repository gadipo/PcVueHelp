BUFTOEXCEL Example

Applies to

This program illustrates the use of the BUFTOEXCEL instruction. Three functions are included, one for each of the modes, CLEARSHEET, MERGE and APPEND.

Example prerequisites

  • The example uses a text file called names.txt as the source which must be located in the project's TP folder. The following is an example of the contents of the text file.

    names.txt
    john,smith
    		 
    joe,bloggs
    		 
    any,body
    		 
    andy,smith
    		 
    adrian,mole
    		 
SUB BUFTOEXEL_CLEARSHEET()
 
'-------- Variables used by BUFTOEXCEL
 
  DIM sWorkBookPath AS STR, sSheetName AS STR;
 
  DIM sColumnSeparator AS STR, sLineSeparator AS STR;
 
  DIM sFileMode AS STR;
 
  DIM iReturn AS INTEGER;
 
'------- Variables used by FILETOBUF  
 
  DIM sFileName AS STR;
 
  DIM lHandle AS LONG;
 
  Const BUFFERSIZE = 1024;
 
'-------- Initialize variables
 
  sFileName = "names.txt";
 
  sWorkBookPath = "names.xlsx"; 'File in the default project directory TP
 
  sColumnSeparator = ",";
 
  sLineSeparator = "\n";
 
  sFileMode = "CLEARSHEET"; 'Creates a new book and worksheet or wipes clean if it already exsists
 
  sSheetName = "SHEET1";
 
'-------- Create a buffer using contents of file
 
  lHandle = FILETOBUF(sFileName);
 
'-------- Create spreadsheet
 
  iReturn = BUFTOEXCEL(lHandle, sLineSeparator, sColumnSeparator, sWorkBookPath, sSheetName, sFileMode);
 
  PRINT("Return from BUFTOEXCEL = ", iReturn);
 
  FREE_BUFFER(lHandle);
 
END SUB
 
 
 
SUB BUFTOEXEL_MERGE()
 
'-------- Variables used by BUFTOEXCEL
 
  DIM sWorkBookPath AS STR, sSheetName AS STR;
 
  DIM sColumnSeparator AS STR, sLineSeparator AS STR;
 
  DIM iOffsetLineNumber AS INTEGER, iOffsetColumnNumber AS INTEGER;
 
  DIM sFileMode AS STR;
 
  DIM iReturn AS INTEGER;
 
'------- Variables used by FILETOBUF  
 
  DIM sFileName AS STR;
 
  DIM lHandle AS LONG;
 
  Const BUFFERSIZE = 1024;
 
'-------- Initialize variables
 
  sFileName = "names.txt";
 
  sWorkBookPath = "names.xlsx"; 'File in the default project directory TP
 
  sColumnSeparator = ",";
 
  sLineSeparator = "\n";
 
  sFileMode = "MERGE"; 'Merges to existing workbook and sheet
 
  sSheetName = "SHEET1";
 
  iOffsetLineNumber = 10;
 
  iOffsetColumnNumber = 10;
 
'-------- Create a buffer using contents of file
 
  lHandle = FILETOBUF(sFileName);
 
'-------- Create spreadsheet
 
  iReturn = BUFTOEXCEL(lHandle, sLineSeparator, sColumnSeparator, sWorkBookPath, sSheetName, sFileMode, iOffsetLineNumber, iOffsetColumnNumber);
 
  PRINT("Return from BUFTOEXCEL = ", iReturn);
 
  FREE_BUFFER(lHandle);
 
END SUB
