JSONPATH Example

Applies to

This program illustrates the use of the JSONPATH instruction.

Example prerequisites

  • A Json formatted file, example.json, located in the project's TP folder. The file contents is as follows.

example.json
{ 
 
"stores":[ 
 
    "Lambton Quay", 
 
    "Willis Street" 
 
], 
 
"manufacturers":[ 
 
{ 
 
    "name":"Acme Co", 
 
    "products":[ 
 
    { 
 
    "name":"Anvil", 
 
    "price":50 
 
    } 
 
   ] 
 
}, 
 
{ 
 
    "name":"Contoso", 
 
    "products":[ 
 
    { 
 
    "name":"Elbow Grease", 
 
    "price":99.95 
 
    }, 
 
    { 
 
    "name":"Headlight Fluid", 
 
    "price":4 
 
    } 
 
   ] 
 
  } 
 
 ]  
 
} 

 

Exec_JSONPATH.svb
Sub Main ()
 
 
 
  DIM iReturn AS INTEGER;DIM i As INTEGER;
 
  DIM sJsonFileName AS STR, sNameSpace AS STR, sJsonPath AS STR;
 
  DIM sReturn AS STR;
 
'------ Initialise variables
 
  sJsonFileName = "example.json";
 
  sNameSpace = "MyNameSpace";
 
'-------- Load JSON formatted file
 
  iReturn = JSONPATH("LOADFILE", sNameSpace, sJsonFileName);
 
  PRINT("Result from JSONPATH LOADFILE = ", iReturn);
 
'-------- Loop to extract JSON values
 
  FOR(i=0; i<2; i++)
 
    sJsonPath = FORMAT("manufacturers[%d].name", i);
 
    sReturn = JSONPATH("GETSTR", sNameSpace, sJsonPath);
 
    PRINT("Result from JSONPATH GETSTR ", sJsonPath, " is ", sReturn);
 
 
 
    sJsonPath = FORMAT("manufacturers[%d].products[0].name", i);
 
    sReturn = JSONPATH("GETSTR", sNameSpace, sJsonPath);
 
    PRINT("Result from JSONPATH GETSTR ", sJsonPath, " is ", sReturn);
 
 
 
    sJsonPath = FORMAT("manufacturers[%d].products[0].price", i);
 
    iReturn = JSONPATH("GETINT", sNameSpace, sJsonPath);
 
    PRINT("Result from JSONPATH GETINT ", sJsonPath, " is ", iReturn);
 
  NEXT
 
'-------- Release resources  
 
  iReturn = JSONPATH("UNLOAD", sNameSpace);
 
  PRINT("Result from JSONPATH UNLOAD = ", iReturn);
 
End Sub
