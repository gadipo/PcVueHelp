CAMERADISPLAY Example

Applies to

This program illustrates the use of the CAMERADISPLAY instruction.

Example prerequisites

  • A mimic containing an ONVIF control and a List-box control must be open. Show picture

  • One text variable called @SB_EXAMPLES.PresetName is required.

'-------- Program wide variables
 
DIM sWindow AS STR, sBranch AS STR, sIdentity AS STR;
 
DIM sAIListBox AS STR;
 
 
 
'-------- Must be ran first to initialize program wide variables  
 
SUB Main ()
 
  sWindow = "Mimic2"; 'Name of the mimic containing the ONVIF control
 
  sBranch = "";
 
  sIdentity = "SecDevice1"; 'Identity of the ONVIF camera
 
  sAIListBox = "AIListBox1"; 'Identity of the list box used to display the camera presets
 
  Exec_GetDeviceInfo();
 
END SUB  
 
 
 
'-------- Get device info and display in results window  
 
SUB Exec_GetDeviceInfo () 
 
  DIM sDevInfoName AS STR, sResult AS STR;
 
  sDevInfoName = "Model";
 
  sResult = CAMERADISPLAY("GETDEVICEINFO", sWindow, sBranch, sIdentity, sDevInfoName);
 
  PRINT("Device model ", sResult);
 
  sDevInfoName = "Firmware";
 
  sResult = CAMERADISPLAY("GETDEVICEINFO", sWindow, sBranch, sIdentity, sDevInfoName);
 
  PRINT("Device firmware ", sResult);
 
  sDevInfoName = "Hardware";
 
  CAMERADISPLAY("GETDEVICEINFO", sWindow, sBranch, sIdentity, sDevInfoName);
 
  PRINT("Device hardware version ", sResult);
 
  sDevInfoName = "Serial";
 
  CAMERADISPLAY("GETDEVICEINFO", sWindow, sBranch, sIdentity, sDevInfoName);
 
  PRINT("Device serial number ", sResult);
 
END SUB
 
 
 
'-------- Get camera presets and display in a list box
 
SUB Exec_GetPresets()
 
  DIM iPresetCount AS INTEGER, iIndex AS INTEGER;
 
  DIM sText AS STR, sUserData AS STR;
 
  LISTBOX("CLEAR", sWindow, sBranch, sAIListBox);
 
  iPresetCount = TOI(CAMERADISPLAY("GETPRESETCOUNT", sWindow, sBranch, sIdentity));
 
  For (iIndex=0; iIndex < iPresetCount; iIndex++)
 
    sText = CAMERADISPLAY("GETPRESETNAME", sWindow, sBranch, sIdentity, iIndex);
 
    sUserData = sText;
 
    LISTBOX("INSERT", sWindow, sBranch, sAIListBox, sText, sUserData, iIndex);
 
  Next
 
END SUB
 
 
 
'-------- Go to a preset. Preset name comes from the list box populated by Exec_GetPresets
 
SUB Exec_GoToPreset()
 
  DIM sPresetName AS STR;
 
  DIM iIndex AS INTEGER;
 
  iIndex = TOI(LISTBOX("GETSELECTEDINDEX", sWindow, sBranch, sAIListBox));
 
  sPresetName = LISTBOX("GETUSERDATA", sWindow, sBranch, sAIListBox, iIndex);
 
  CAMERADISPLAY("GOTOPRESET", sWindow, sBranch, sIdentity, sPresetName);
 
END SUB
 
 
 
'-------- Go to home
 
SUB Exec_GoToHome()
 
  CAMERADISPLAY("GOTOHOME", sWindow, sBranch, sIdentity);
 
END SUB
 
 
 
'-------- Pan the camera. Direction (LEFT or RIGHT) passed as an argument to the function
 
SUB Exec_Pan ()
 
  DIM cDirection AS STR;
 
  cDirection = GETARG("ARG1");
 
  CAMERADISPLAY("PAN", sWindow, sBranch, sIdentity, cDirection);
 
END SUB
 
 
 
'-------- Tilt the camera. Direction (UP or DOWN) passed as an argument to the function
 
SUB Exec_Tilt ()
 
  DIM cDirection AS STR;
 
  cDirection = GETARG("ARG1");
 
  CAMERADISPLAY("TILT", sWindow, sBranch, sIdentity, cDirection);
 
END SUB
 
 
 
'-------- Zoom the camera. Direction (IN or OUT) passed as an argument to the function
 
SUB Exec_Zoom ()
 
  DIM cDirection AS STR;
 
  cDirection = GETARG("ARG1");
 
  CAMERADISPLAY("ZOOM", sWindow, sBranch, sIdentity, cDirection);
 
END SUB
 
 
 
'-------- Start the camera's guard tour
 
SUB Exec_StartGuardTour()
 
  DIM iDelay AS INTEGER;
 
  iDelay = 10;
 
  CAMERADISPLAY("STARTGUARDTOUR", sWindow, sBranch, sIdentity, iDelay);
 
END SUB
 
 
 
'-------- Stop the camera's guard tour
 
SUB Exec_StopGuardTour ()
 
  CAMERADISPLAY("STOPGUARDTOUR", sWindow, sBranch, sIdentity);
 
END SUB
 
 
 
'-------- Add a preset and then get the new list of presets. The new preset name is supplied by the variable @SB_EXAMPLES.PresetName
 
SUB Exec_AddPreset()
 
  DIM sPresetName AS STR;
 
  sPresetName = @SB_EXAMPLES.PresetName;
 
  CAMERADISPLAY("ADDPRESET", sWindow, sBranch, sIdentity, sPresetName);
 
  Exec_GetPresets();
 
END SUB
 
 
 
'-------- Remove the preset name supplied by the variable @SB_EXAMPLES.PresetName
 
SUB Exec_RemovePreset()
 
  DIM sPresetName AS STR;
 
  sPresetName = @SB_EXAMPLES.PresetName;
 
  CAMERADISPLAY("REMOVEPRESET", sWindow, sBranch, sIdentity, sPresetName);
 
  Exec_GetPresets();
 
END SUB
