Example 1 - Extract the value of an Xml element from an Xml file

Contents of the file test.xml

<?xml version="1.0" encoding="UTF-8"?>
 
 <xmlpathtest>
 
    <test>1</test>
 
    <stringtest>Hello</stringtest>
 
</xmlpathtest>

Sample script to extract the value from an Xml element

Sub Main()
 
 
 
  Dim cXPath As Str, cReturn As Str;
 
  Dim iReturn As Integer, i As Integer;
 
  Dim sReturn As Single;
 
 
 
  iReturn = XMLPATH ( "LOADFILE", "CTX1", "test.xml" );
 
  Print ("Return from LOADFILE = ", iReturn);
 
  cXPath = "xmlpathtest[1]/test[1]";
 
  
 
  iReturn = XMLPATH ( "GETINT", "CTX1", cXPath);
 
  Print("GETINT for ", cXPath, " = ", iReturn);
 
  cXPath = "xmlpathtest[1]/stringtest[1]";
 
  
 
  cReturn = XMLPATH ( "GETSTR", "CTX1", cXPath);
 
  Print("GETSTR for ", cXPath, " = ", cReturn);
 
 
 
End Sub
 
 

Example 2 - Extracting more values from an Xml file

Contents of the file books.xml

<?xml version="1.0" encoding="UTF-8"?>
 
  <bookstore>
 
    <book category="Cooking">
 
     <title>Everyday Italian</title>
 
      <details lang="en" author="Giada De Laurentiis" year="2005" price="30.99" />
 
    </book>
 
    <book category="Children">
 
    <title>Harry Potter</title>
 
      <details lang="en" author="J K. Rowling" year="2005" price="5.99" />
 
    </book>
 
    <book category="Web">
 
    <title>XQuery Kick Start</title>
 
      <details lang="en" author="James McGovern" year="2003" price="49.99" />
 
    </book>
 
  </bookstore>

Sample script to extract more values

Sub Main()
 
  Dim cXPath As Str, cReturn As Str;
 
  Dim iReturn As Integer, i As Integer;
 
  Dim sReturn As Single;
 
 
 
  iReturn = XMLPATH ( "LOADFILE", "CTX1", "title.xml" );
 
  Print ("Return from LOADFILE = ", iReturn);
 
 
 
  For (i=1; i<4; i++)
 
    cXPath = FORMAT ("bookstore/book[%1].category", i);
 
    cReturn = XMLPATH ( "GETSTR", "CTX1", cXPath);
 
    Print(cXPath, " = ", cReturn);
 
 
 
    cXPath = FORMAT ("bookstore/book[%1]/title", i);
 
    cReturn = XMLPATH ( "GETSTR", "CTX1", cXPath);
 
    Print(cXPath, " = ", cReturn);
 
 
 
cXPath = FORMAT ("bookstore/book[%1]/details.author", i);
 
    cReturn = XMLPATH ( "GETSTR", "CTX1", cXPath);
 
    Print(cXPath, " = ", cReturn);
 
 
 
    cXPath = FORMAT ("bookstore/book[%1]/details.year", i);
 
    iReturn = XMLPATH ( "GETINT", "CTX1", cXPath);
 
    Print(cXPath, " = ", iReturn);
 
 
 
    cXPath = FORMAT ("bookstore/book[%1]/details.price", i);
 
    sReturn = XMLPATH ( "GETSINGLE", "CTX1", cXPath);
 
    Print(cXPath, " = ", sReturn);
 
  Next
 
 
 
  iReturn = XMLPATH ( "UNLOAD", "CTX1");
 
  Print ("Return from UNLOAD = ", iReturn);
 
End Sub
