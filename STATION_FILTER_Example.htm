STATION_FILTER Example

Applies to

This example filters alarm viewers so that the currently logged user only accesses alarms matching the populations PopuDomain01 and PopuDomain02.

SUB PopuDomain01or02()

'Declare

DIM hBuffer as Long;

DIM strPopu as Str;

DIM intLength as Integer;

hBuffer = ALLOC_BUFFER(255);

strPopu = "FILTER,0,1,PopuDomain01,PopuDomain02";

PUT_BUFFER(hBuffer, 0, strPopu);

Station_filter("APPLY", hBuffer);

FREE_BUFFER(hBuffer);

END SUB
